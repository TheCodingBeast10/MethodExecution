# MethodExecution
public class Main {

	public static void main(String[] args) {
          double sum1= CalculateSum(2.45,3.55);
	    System.out.println(sum1);
	    
	    double sum2=CalculateSum(3,4);
	    System.out.println(sum2);
                  
	}
public static double CalculateSum(double a,double b){
	    double sum= a+b;
	    return sum;
	}        
	
}
